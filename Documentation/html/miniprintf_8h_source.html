31 // From the README file:
34 
35 Tested and estimated to require about 640 bytes of code for STM32F103C8T6. Should
36 be usable on any MCU platform that supports:
37 
38  #include <stdarg.h>
39  #include <string.h>
40 
41 SUPPORT:
42 
43 Each format item %+0wd, %0wx and for strings %-ws, the following
44 applies:
45 
46  + Optional: Indicates sign should always print (d and x)
47  - Optional: Indicates field should be left justified (s)
48  0 Optional: Pad with leading zeros (d and x)
49  w Optional: Decimal field width
50 
51  Formats %c, %d, %x and %s are supported (only). '%%' prints as '%'.
52 
53  Floating point is not supported, keeping this library minimal.
54 
55 FORMAT EXAMPLES:
56 
57  %+05d '+00009' int is 9.
58  %d '9'
59  %03d '009'
60  %04x '001F' int is 31
61  %x '1F'
62  %-9s 'abc ' string was 'abc'
63  %9s ' abc'
64  %s 'abc'
65 
66 STRING FORMATTING:
67 
68  int mini_snprintf(char *buf, unsigned maxbuf, const char *format, ...);
69 
70  See standard snprintf(3). Note that the output is null terminated
71  when the buffer size permits.
72 
73 DEVICE FORMATTING HOWTO:
74 
75  int mini_vprintf_cooked(void (*putc)(char), const char *format, va_list args);
76  int mini_vprintf_uncooked(void (*putc)(char), const char *format, va_list args);
77 
78  (0) Decide: cooked or uncooked output?
79 
80  COOKED means that a CR is sent after every LF is sent out,
81  like UNIX terminal output.
82 
83  UNCOOKED means no CR processing is performed. Like snprintf,
84  what you format is what you get.
85 
86  (1) Declare your own putc function, something like:
87 
88  static void uart_putc(char ch) {
89  usart_send_blocking(USART1, ch); // libopencm3
90  }
91 
92  (2) Declare your own printf function:
93 
94  int uart_printf(const char *format, ...)
95  __attribute((format(printf, 1, 2)));
96 
97  int uart_printf(const char *format, ...) {
98  va_list args;
99  int rc;
100 
101  va_start(args, format);
102  rc = mini_vprintf_cooked(uart_putc, format, args);
103  va_end(args);
104  return rc;
105  }
106 
107  The attribute clause is optional, but when provided can only
108  appear in the function prototype. It allows the compiler to
109  check that you have appropriate arguments for each format item.
110 
111  (3) Use it:
112 
113  int flea_count = 45;
114 
115  uart_printf("My dog has %d fleas.\n", flea_count);
116 
117 NOTES:
118  1. Stack usage is minimal (perhaps 256 bytes).
119  2. No malloc/realloc/free calls (no heap usage)
120  3. Re-entrant (no static storage used)
121  4. Compromizes favoured smaller code over speed.
